# Knapsack HC

from random import choice
from typing import List, Dict, Tuple


def hill_climbing_search(problem, search_iterations_limit=200) -> Tuple[ List[bool], float ]:
    # 1. Initialize Solution
    solution, solution_fitness = initialize_solution(problem.objective_function, problem.is_admissible, problem.decision_variables, problem.constraints)
    
    # 2. REPEAT (loop)
    for i in range(search_iterations_limit):
        # 2.1. 
        best_neighbor, best_neighbor_fitness = neighborhood_function(solution, problem.objective_function, problem.is_admissible, problem.decision_variables, problem.constraints)
        if best_neighbor_fitness >= solution_fitness:
            solution = best_neighbor
            solution_fitness = best_neighbor_fitness
        else:
            break
    
    # 3. Return solution and solution's fitness
    return solution, solution_fitness


def initialize_solution(objective_function, is_admissible, decision_variables={}, constraints={}) -> Tuple[ List[bool], float ]:
    admissible = False
    while not admissible:
        solution = build_solution(decision_variables)
        admissible = is_admissible(solution, decision_variables, constraints)
    fitness = objective_function(solution, decision_variables)
    return solution, fitness


def build_solution(decision_variables) -> List[bool]:
    solution = []
    possible_values = [True, False]
    for i in range(len(decision_variables["values"])):
        randonly_selected_value = choice(possible_values)
        solution.append(randonly_selected_value)
    return solution


def neighborhood_function(solution, objective_function, is_admissible,  decision_variables={}, constraints={}) -> Tuple[ List[bool], float ]:
    neighborhood = []
    for i in range(len(solution)):
        neighbor = generate_neighbor(solution, i)
        if is_admissible(neighbor, decision_variables, constraints):
            neighborhood.append(neighbor)
    best_neighbor, best_neighbor_fitness = get_best_neighbor(neighborhood, objective_function, decision_variables)
    return best_neighbor, best_neighbor_fitness


def generate_neighbor(solution, i) -> List[bool]:
    neighbor = solution.copy()
    neighbor[i] = not neighbor[i]
    return neighbor


def get_best_neighbor(neighborhood, objective_function, decision_variables={}) -> Tuple[ List[bool], float ]:
    fitnesses = []
    for neighbor in neighborhood:
        neighbor_fitness = objective_function(neighbor, decision_variables)
        fitnesses.append(neighbor_fitness)
    best_neighbor_fitness = max(fitnesses)
    best_neighbor_index = fitnesses.index(best_neighbor_fitness)
    best_neighbor = neighborhood[best_neighbor_index]
    return best_neighbor, best_neighbor_fitness
