# Problem: Knapsack / Algorithm: Simulated Annealing

from random import choice, random
from math import exp


# PROBLEM REPRESENTATION -----------------------------------//
decision_variables = {
    "values"  : [2, 4,  10, 2, 1],
    "weights" : [1, 12, 4,  2, 1]
}

# solution encoding -> e.g. [True, False, True, True, False]

constraints = {
    "max_weight": 15
}
# ----------------------------------------------------------//


# SIMULATED ANNEALING PARAMETERS ---------------------------//
sa_parameters = {
    "control_parameter": 1.0,        # C
    "control_minimum": 0.5,          # C minimum (limit)
    "decrease_rate": 0.95,           # value by which C is decreased
    "internal_loop_iterations": 100  # L
}
# ----------------------------------------------------------//


def simulated_annealing_search(decision_variables={}, constraints={}, sa_parameters={}):

    # 1 & 2. Initialize Simulated Annealing parameters — RECEIVED AS ARGUMENT:
        #  SA Parameters:
            # control_parameter: 1.0
            # control_limit: 0.5
            # decrease_rate: 0.95
            # internal_loop_iterations: 100

    # 3. Initialize random solution X
    solution, solution_fitness = initialize_solution(decision_variables, constraints)

    # 4. Repeat until termination condition
    while sa_parameters["control_parameter"] >= sa_parameters["control_minimum"]:
        
        # 4.1. Repeat L times
        for i in range(sa_parameters["internal_loop_iterations"]):

            # 4.1.1. Generate a RANDOM neighbor Y from the Neighborhood of X
            neighborhood = neighborhood_function(solution, decision_variables, constraints)
            random_neighbor, random_neighbor_fitness = get_random_neighbor(neighborhood, decision_variables)
            
            # 4.1.2. If fitness(Y) is greater or equal to fitness(X)
            switch_solutions = False
            if random_neighbor_fitness >= solution_fitness:
                # Then X := Y
                switch_solutions = True
            # Else
            else:
                # If acceptance probability > random[0,1)
                if probability_test(solution_fitness, random_neighbor_fitness, sa_parameters):
                    # Then X := Y
                    switch_solutions = True
            
            if switch_solutions:    
                solution = random_neighbor
                solution_fitness = random_neighbor_fitness
        
        # 4.2. Update C (control parameter)
        sa_parameters["control_parameter"] *= sa_parameters["decrease_rate"]
    
    # 5. Return X
    return solution, solution_fitness


# ------------------------------------------------------//
# BUILDING BLOCKS:
# ------------------------------------------------------//
def initialize_solution(decision_variables={}, constraints={}):
    admissible = False
    while not admissible:
        solution = build_solution(decision_variables)
        admissible = is_admissible(solution, decision_variables, constraints)
    fitness = objective_function(solution, decision_variables)
    return solution, fitness


def build_solution(decision_variables): 
    solution_size = len(decision_variables["values"])
    solution = []
    possible_values = [True, False]
    for i in range(solution_size):
        randomly_selected_value = choice(possible_values)
        solution.append(randomly_selected_value)
    return solution


def is_admissible(solution, decision_variables={}, constraints={}):
    solution_weight = 0
    solution_size = len(decision_variables["weights"])
    for i in range(solution_size):
        if solution[i]:
            solution_weight += decision_variables["weights"][i]
    return solution_weight < constraints["max_weight"]

  
def objective_function(solution, decision_variables={}):
    fitness = 0
    solution_size = len(decision_variables["values"])
    for i in range(solution_size):
        if solution[i]:
            fitness += decision_variables["values"][i]
    return fitness


# ------------------------------------------------------//
# Algorithm's functions for navigating the Search Space:
# ------------------------------------------------------//
def neighborhood_function(solution, decision_variables={}, constraints={}):
    neighborhood = []
    for i in range(len(solution)):
        neighbor = generate_neighbor(solution, i)
        if is_admissible(neighbor, decision_variables, constraints):
            neighborhood.append(neighbor)
    return neighborhood


def generate_neighbor(solution, i):
    # bit flip
    neighbor = solution.copy()
    neighbor[i] = not neighbor[i]
    return neighbor


def get_random_neighbor(neighborhood, decision_variables):
    random_neighbor = choice(neighborhood)
    random_neighbor_fitness = objective_function(random_neighbor, decision_variables)
    return random_neighbor, random_neighbor_fitness


def probability_test(solution_fitness, neighbor_fitness, sa_parameters):
    probability_of_accepting = exp(-1 * (abs(solution_fitness - neighbor_fitness) / sa_parameters["control_parameter"]))
    if probability_of_accepting > random():
        return True
    return False

# ---------------------------------------------//
# ---------------------------------------------//


# Run search and print result (i.e. chosen solution)
solution, solution_fitness = simulated_annealing_search(decision_variables, constraints, sa_parameters)

print(f"solution: {solution}")
print(f"solution's fitness: {solution_fitness}")
