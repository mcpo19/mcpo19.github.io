# Coding exercises and solutions

# Exercise 1

Define a function called find_max_value that receives a list of integers as argument and returns its highest value.

# Test it with the following code:
integer_list = [52, 81, 33, 17, 25, 90, 28, 43]
max_value = find_max_value(integer_list)
print(max_value)

Expected output is 90.

# Solutions exercise 1

# SOLUTION 1.1. – with for each loop:

def find_max_value(list_arg):
    item_to_compare = list_arg[0]
    for item in list_arg[1:-1]:
        if item > item_to_compare:
            item_to_compare = item
    return item_to_compare

# SOLUTION 1.2. – with for i loop (and range() function):

def find_max_value(list_arg):
    item_to_compare = list_arg[0]
    for i in range(1, len(list_arg)):
        if list_arg[i] > item_to_compare:
            item_to_compare = list_arg[i]
    return item_to_compare

# SOLUTION 1.3. – with Python's built-in function max():

def find_max_value(list_arg):
    return max(list_arg)

# Exercise 2

Define a function called find_item_index that takes a list of integers and a value as arguments. If the value is in the list, return its position in the list (i.e. index). If its not, return a feedback message to the user.

# Code for testing:
integer_list = [52, 81, 33, 17, 25, 90, 28, 43]

# 1st test:
value_searched = 17
result = find_item_index(integer_list, value_searched)
print(result)

# 2nd test:
value_searched = 35
result = find_item_index(integer_list, value_searched)
print(result)

# Solutions exercise 2

# SOLUTION 2.1. - with for i loop (and range() function):

def find_item_index(list_arg, value):
    for i in range(len(list_arg)):
        if list_arg[i] == value:
            return i
    return "Value is not in the list."

# SOLUTION 2.2. – with for each loop and list's .index() method:

def find_item_index(list_arg, value):
    for item in list_arg:
        if item == value:
            index = list_arg.index(item)
            return index
    return "Value is not in the list."

# Exercise 3

Using Python's utility function input(<message to user>) write a program that accepts a user's string input and outputs its inverse. Example of expected output:

[In] : anagram
[Out]: margana

# Solutions exercise 3

# SOLUTION 3.1. - with for each loop

def reverse_user_string():
    user_string = input("Write Something \n")
    reversed_string = "" 
    for letter in user_string: 
        reversed_string = letter + reversed_string
    return reversed_string

Checking result:

# SOLUTION 3.2. - with Python's slices

Slices syntax –> [ starting_index : stop_index : indices_to_step ]

def reverse_user_string():
    user_string = input("Write Something \n")
    reversed_string = user_string[::-1]
    return reversed_string

# SOLUTION 3.3. - with Python's built-in reversed() function

def reverse_user_string():
    user_string = input("Write Something \n")
    reversed_string = "".join(reversed(user_string)) 
    return reversed_string

# Exercise 4

Build a Pyramid! Create a program that accepts an integer from the user and builds an equivalent number of layers, using #. The idea is to simulate a pyramid. Example of expected output:

[In] : 8
[Out]:
#
##
###
####
#####
######
#######
########

# Solution exercise 4

def create_pyramid_iterative(pyramid_height):
    pyramid_print = ""
    for i in range(pyramid_height):
        layer = ["#" for _ in range(i+1)]
        pyramid_print += "".join(layer)
        pyramid_print += "\n"
    print(pyramid_print)


# Testing:
create_pyramid_iterative(20)

# Exercise 5

Define a function called find_item_index_fast that takes a list of integers and a value as arguments. The objective is the same as in Exercise 2, only now you should aim to accomplish the search in a more efficient way. Independent of the input size (i.e. the list's size). Tip: Notice this time the example integer_list is ordered...

# Code for testing:
integer_list = [1, 2, 11, 12, 15, 17, 25, 26, 28, 28, 29, 33, 43, 44, 47, 50, 52, 61, 73, 81, 82, 90, 99, 100]

# 1st test:
value_searched = 81
result = find_item_index_fast(integer_list, value_searched)
print(result)

# 2nd test:
value_searched = 30
result = find_item_index_fast(integer_list, value_searched)
print(result)

# Solution exercise 5

The idea here was to implement the Binary Search algorithm.

def find_item_index_fast(items_list, searched_item):
    first = 0
    last = len(items_list) - 1
    midpoint = None
    found = False
    while first <= last and not found:
        midpoint = (first + last) // 2
        if items_list[midpoint] == searched_item:
            found = True
        else:
            if searched_item < items_list[midpoint]:
                last = midpoint - 1
            else:
                first = midpoint + 1
    if found:
        return midpoint
    return "Value not in list."

# Exercise 6

Build your very own factorial calculator. The program should take an integer as argument and output its factorial. Try to find two different solutions to this problem.

# Solutions exercise 6

# Solution 6.1. - with for i loop.

def factorial_with_loop(value):
    result = 1
    for i in range(1, value + 1): 
        result *= i
    return result


# Test solution:
print(factorial_with_loop(6))

# Solution 6.2. - with recursive method (calls the function itself).

def factorial_with_recursion(value):
    if value <= 1:
        return 1
    else:
        return value * factorial_with_recursion(value-1)


# Test solution:
print(factorial_with_recursion(6))

# 7.2. Draw arbitrary shape

def draw_shape(number_of_vertices):
    turn_calc = 360 / number_of_vertices
    while True:
        forward(100)
        left(turn_calc)
        if abs(pos()) < 1:
            break
    done()

draw_shape(10)
